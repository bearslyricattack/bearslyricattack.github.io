Floyd最短路算法

发表于 2022-10-17 分类于 笔记 
#include<stdio.h>

​
int main (){
//首先,这个社交网络图很明显,是一张无向图.用邻接矩阵进行存储和遍历.
//确定邻接矩阵的大小.
int size,length;
printf(“请输入社交网络图的节点数:\n”);
scanf(“%d”,&size);
printf(“请输入社交网络图的边数:\n”);
scanf(“%d”,&length);
int zu[size][size];
int i,j,k;

//初始值为0 
for(i=0;i<size;i++){
    for(j=0;j<size;j++){
        if(i==j){
            zu[i][j]=0;
        } else
        {
            zu[i][j]=999;
        } 
    } 
} 

int p1,p2; 
//为邻接矩阵赋值
for(i=0;i<length;i++){
    printf("请输入直接联通的两个节点的编号:\n");
    scanf("%d %d",&p1,&p2);
    zu[p1-1][p2-1]=1;
    zu[p2-1][p1-1]=1; 
} 

//输出邻接矩阵,做测试用.
for(i=0;i<size;i++){
    for(j=0;j<size;j++){
        printf("%d ",zu[i][j]); 
    } 
    printf("\n"); 
} 	

//具体操作的思路是: 对每个节点执行: 找出与其相连的节点,加入一个一位数组中,
//同时把这几个节点的标号记录到另一个一维度数组中.
//然后再对第二个数组中的每个节点执行相同的操作,所有的节点都汇总到第一个数组中.
//重复进行五次,最后根据第一个数组的结果判断与这个节点距离不超过6的节点. 
//然后发现基本操作是:输入一个节点和一个数组,遍历之后得到一个临时一维数组,这个函数设置为base()


//..... 
//当抽象完函数的时候,就会发现这也太复杂了,有没有什么简单的方法呢?

//Floyd最短路算法:把这个问题转化为两点之间最短距离的问题(任意两点间只要联通,就设置距离为1)

//核心代码
for(int k = 0 ; k < size ; k ++)
{
    for(int i = 0 ; i < size ; i ++)
    {
        for(int j = 0 ; j < size ; j ++)
        {
            if(zu[i][j] > zu[i][k] + zu[k][j]){
            	zu[i][j] = zu[i][k] + zu[k][j];
            } 
        }
    }
}	 

//现在,每个位置上的数已经变成了这两个点之间的距离,如果这两个点不连通,那就还是为0;
printf("\n\n\n"); 

printf("现在 最短路径的邻接矩阵为:\n");
//输出邻接矩阵,做测试用.
for(i=0;i<size;i++){
    for(j=0;j<size;j++){
        printf("%5d ",zu[i][j]); 
    } 
    printf("\n"); 
} 

printf("\n\n\n"); 	

int count;
for(i=0;i<size;i++){
    count=0;
    for(j=0;j<size;j++)
    {
        if(zu[i][j]<=6)
        {
            count++;
        }
    }
    printf("%d: %.2f%%\n",i+1,(float)count/size);
}  
}
